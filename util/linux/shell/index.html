    if

    #!/bin/bash

# if后面有空格
# [] 里面各个符号之间的空格
if [ 121 -eq 1 ]; then
    echo 'bb'
elif [ 1 -le 2 ]; then
    echo 'aaa'
fi

if [ 'aaaa' == 'aa' ]; then
    echo '=='
else
    echo '!='
fi

    case

    #!/bin/bash

case $1 in
"1")
    echo "1"
# ;; 相当于break    
;;
"2")
    echo "2"
;;
# * 相当于default
*)
    echo default
;;
esac

sum=0

for i in $@

do
    echo "$i"

for j in "$@"
do

    $*和$@都表示传递给函数或脚本的所有参数，不被双引号“”包含时，都以$1 $2 …$n的形式输出所有参数

    当它们被双引号“”包含时，“$*”会将所有的参数作为一个整体，以“$1 $2 …$n”的形式输出所有参数；“$@”会将各个参数分开，以“$1” “$2”…”$n”的形式输出所有参数。

    while

    #!/bin/bash


i=1
while [ $i -lt 10 ]
do
    echo $i
    i=$[$i+1]
done

    自定义函数

    #!/bin/bash

function sum()
{
    s=0
    s=$[$1+$2]
    echo $s
    $?
}
# 需要指定参数
sum $1 $2
if [ 0 -eq 0 ]
then
    echo 111
fi
